Example: A story routing capability query
-- "Implement subscription billing endpoint"
story.engg_domain    = 'backend'
story.org_domain     = 'monetization'
story.industry      = 'ott'   ← inherited from workspace

-- Routing query against capability_ratings
SELECT agent, model_version, score FROM capability_scores
WHERE engg_domain = 'backend'
  AND org_domain  = 'monetization'
  AND industry   = 'ott'
  AND phase      = 'build'
ORDER BY score DESC;
